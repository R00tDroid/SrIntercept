file(GLOB_RECURSE SourceFiles
    ${CMAKE_CURRENT_LIST_DIR}/Source/*.*
)

add_executable(Manager WIN32
    ${SourceFiles}
)

    Detours
    Shared
    Softcam
    ImGui
    d3d11
)

    ${CMAKE_CURRENT_LIST_DIR}/Source
    ${detours_SOURCE_DIR}/src
)
